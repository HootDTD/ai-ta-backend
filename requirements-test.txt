# Test-only dependencies for the AI-TA backend.

#

# Approved for Phase 1 of docs/TESTING-CI-PLAN.md (full set).

# --- core test runner plugins ---------------------------------------------
pytest>=8.0              # the runner itself — was only transitive (via -asyncio/-cov); pin explicitly
pytest-cov>=5.0          # coverage measurement + gating (diff-cover feeds off this)
pytest-mock>=3.14        # mocker fixture (thin wrapper over unittest.mock)
pytest-xdist>=3.6        # parallel test execution (-n auto)
pytest-timeout>=2.3      # hard per-test timeouts (LLM-path tests must be sub-second)
pytest-randomly>=3.15    # randomize test order to surface inter-test coupling

vcrpy>=6.0               # record/replay HTTP cassettes for integration LLM calls

freezegun>=1.5           # freeze time in tests

factory-boy>=3.3         # model factories (tests/factories/)

# --- real Postgres + pgvector test harness --------------------------------
testcontainers[postgres]>=4.0   # ephemeral pgvector/pgvector:pg16 container per session

# --- coverage gating / quality -------------------------------------------
diff-cover>=9.0          # patch-coverage gate vs origin/staging
ruff==0.15.22            # lint + format (pinned: CI format --check must match the version the code was formatted with)
